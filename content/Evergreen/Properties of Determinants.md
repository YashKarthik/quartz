$$
\begin{align}

\det(kA) = k^n\det(A)\\\\

\det(AB) = (\det A)(\det B)\\\\

\det(A^k) = (\det A)^k\\\\

\det(A^T) = \det(A)\\\\

\det(A^{-1}) = \frac{1}{\det(A)}\\\\

\det(A+B) \ne \det (A) + \det (B)

\end{align}
$$
